import java.util.ArrayList ;

class Solution {
    public ArrayList<Integer> allFactors( int n ) {
        ArrayList<Integer> factors = new ArrayList<Integer>() ;

        for ( int i = 1 ; i*i <= n ; i ++ ){
            if(n % i == 0){
                if( n / i == i ){
                    factors.add( i ) ;
                }
                else{
                    factors.add( i ) ;
                    factors.add( n/i ) ;
                }
            }
        }

        return factors ;
    }

    public int countOfFactors( ArrayList<Integer> factors ) {
        return factors.size() ;
    }

    public int commonFactors(int a, int b) {
        int countOfCommonFactors = 0 ;
        int smallest = a ;
        if ( a > b ) {
            smallest = b ;
        }
        for ( int i = 1 ; i <= smallest ; i ++ ) {
            if ( ( a % i == 0 ) && ( b % i == 0 ) ) {
                countOfCommonFactors ++ ;
            }
        }
        return countOfCommonFactors ;
    }
}
